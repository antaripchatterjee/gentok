typedef myfunc_t :: func(int) bool;

func main(argv string[]) {
    even_odd_finder myfunc_t = @(v i32) -> v % 2 == 0;

    arr := 0 .. 1000; # list of 1001 i32 elements
    soe, soo := for i : arr; even_odd_finder(i); soe := 0, soo := 0 {
        soe += i;
    } else {
        soo += i;
    } finally {
        return soe, soo;
    }
    Printf 'Sum of even numbers is %d\n' soe;
    Printf 'Sum of odd numbers is %d\n' soo;
}
